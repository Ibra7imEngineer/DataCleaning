================================================================================
📝 مثال الاستخدام الكامل
================================================================================

# الخطوة 1: استيراد المحرك
from comprehensive_final_engine import ComprehensiveFinalEngine

# الخطوة 2: إنشاء المحرك
engine = ComprehensiveFinalEngine(verbose=True)

# الخطوة 3: معالجة البيانات
processed_df = engine.process_comprehensive(
    input_path='sample_data.csv',
    output_path='sample_data_processed.xlsx'
)

# الخطوة 4: الوصول للبيانات المعالجة
print(processed_df)

# الخطوة 5: الوصول للسجلات
print("سجل العمليات:")
for entry in engine.audit_log:
    print(entry)

# الخطوة 6: عرض الإحصائيات
print(f"أعمدة التاريخ: {engine.date_columns}")
print(f"الأعمدة الرقمية: {engine.numeric_columns}")

├─ البحث عن الصفوف المكررة
├─ اكتشاف الفراغات العشوائية
├─ توثيق الترحيل
└─ حذف الصفوف المكررة

المرحلة 4️⃣ - تطهير أعمدة التاريخ
├─ فحص كل خلية على حدة

└─ توثيق القيم غير الرقمية

المرحلة 6️⃣ - فحص شامل صف بصف
